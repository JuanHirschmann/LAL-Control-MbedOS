167 void print(const char *str)
168 {
169
170 if (this->line_wrap)
171 {
172 int string_length = strlen(str);
173 int available_rows = LCD_ROWS - this->cursor_position.y_pos;
174 int string_index = 0;
175 while (available_rows > 0)
176 {
177 int available_space = LCD_COLS - this->cursor_position.x_pos;
178 while (available_space > 0 && string_length > 0)
179 {
180 this->send_command(DISPLAY_RS_DATA, str[string_index]);
181 string_index++;
182 string_length--;
183 available_space--;
184 }
185 this->cursor(0, this->cursor_position.y_pos + 1);
186 available_rows--;
187 }
188 }

190 {
191 while (*str)

193 this->send_command(DISPLAY_RS_DATA, *str++);
